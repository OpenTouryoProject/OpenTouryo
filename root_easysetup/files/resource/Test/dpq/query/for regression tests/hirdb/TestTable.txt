-- TABLE: Shippers 
--------------------
CREATE TABLE Shippers(
    ShipperID     INTEGER         NOT NULL,
    CompanyName   NVARCHAR(40)    NOT NULL,
    Phone         NVARCHAR(24),
    PRIMARY KEY (ShipperID)
);

--------------------
-- Sequence: ShipperID
--------------------
CREATE SEQUENCE TS_ShipperID;
