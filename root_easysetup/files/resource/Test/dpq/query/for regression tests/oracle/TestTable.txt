--------------------
-- TABLE: Shippers 
--------------------
CREATE TABLE Shippers(
    ShipperID      NUMBER(10, 0)    NOT NULL,
    CompanyName    NVARCHAR2(40)    NOT NULL,
    Phone          NVARCHAR2(24),
    CONSTRAINT PK_Shippers PRIMARY KEY (ShipperID)
)
;

--------------------
-- Sequence: ShipperID
--------------------
CREATE SEQUENCE TS_ShipperID;
